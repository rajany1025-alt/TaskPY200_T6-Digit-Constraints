# Take input
N = int(input("Enter N: "))

# Assume no number found initially
result = -1

# Loop from 1 to N
for x in range(1, N + 1):

    temp = x
    digit_sum = 0
    count_3 = 0
    contains_zero = False

    # Check digits using while loop
    while temp > 0:

        digit = temp % 10   # get last digit

        # check zero
        if digit == 0:
            contains_zero = True

        # count digit 3
        if digit == 3:
            count_3 = count_3 + 1

        # add to sum
        digit_sum = digit_sum + digit

        temp = temp // 10   # remove last digit

    # check all conditions
    if digit_sum % 7 == 0 and count_3 == 2 and contains_zero == False:
        result = x
        break   # smallest found

# print result
print(result)
